procedure negate_via_loop(int x)
    local int y = 0
    if x > 0 then
        from y = 0 do
            skip
        loop
            x -= 1
            y -= 1
        until x = 0
    else
        from y = 0 do
            skip
        loop
            x += 1
            y += 1
        until x = 0
    fi y < 0
    x <=> y
    delocal int y = 0
