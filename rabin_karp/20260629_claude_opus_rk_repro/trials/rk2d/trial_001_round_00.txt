procedure windowhash(int G[][], int r0, int c0, int m, int d, int h)
  local int i = 0
  from i = 0 do
    local int k = 0
    from k = 0 do
      h *= d
      h += G[r0 + i][c0 + k]
      k += 1
    loop skip until k = m
    delocal int k = m
    i += 1
  loop skip until i = m
  delocal int i = m

procedure countdiff(int T[][], int s1, int s2, int P[][], int m, int diff)
  local int i = 0
  from i = 0 do
    local int k = 0
    from k = 0 do
      if T[s1 + i][s2 + k] != P[i][k] then
        diff += 1
      else
        skip
      fi T[s1 + i][s2 + k] != P[i][k]
      k += 1
    loop skip until k = m
    delocal int k = m
    i += 1
  loop skip until i = m
  delocal int i = m

procedure rabinkarp2(int T[][], int n, int P[][], int m, int d, int q, stack r)
  local int s2 = 0
  from s2 = 0 do
    local int s1 = 0
    from s1 = 0 do
      local int ph = 0
        call windowhash(P, 0, 0, m, d, ph)
        local int th = 0
          call windowhash(T, s1, s2, m, d, th)
          local int phm = ph - (ph / q) * q
            local int thm = th - (th / q) * q
              if thm = phm then
                local int diff = 0
                  call countdiff(T, s1, s2, P, m, diff)
                  if diff = 0 then
                    local int a = s1
                      push(a, r)
                    delocal int a = 0
                    local int b = s2
                      push(b, r)
                    delocal int b = 0
                  else
                    skip
                  fi diff = 0
                  uncall countdiff(T, s1, s2, P, m, diff)
                delocal int diff = 0
              else
                skip
              fi thm = phm
            delocal int thm = th - (th / q) * q
          delocal int phm = ph - (ph / q) * q
          uncall windowhash(T, s1, s2, m, d, th)
        delocal int th = 0
        uncall windowhash(P, 0, 0, m, d, ph)
      delocal int ph = 0
      s1 += 1
    loop skip until s1 = n - m + 1
    delocal int s1 = n - m + 1
    s2 += 1
  loop skip until s2 = n - m + 1
  delocal int s2 = n - m + 1
