procedure compute_V(int A[][], int s1, int s2, int m, int d, int V)
  local int i = 0
  from i = 0 do
    V *= d
    local int rp = 0
      local int k = 0
      from k = 0 do
        rp *= d
        rp += A[s1+i][s2+k]
        k += 1
      loop skip
      until k = m
      delocal int k = m
      V += rp
      local int k2 = m
      from k2 = m do
        k2 -= 1
        rp -= A[s1+i][s2+k2]
        rp /= d
      loop skip
      until k2 = 0
      delocal int k2 = 0
    delocal int rp = 0
    i += 1
  loop skip
  until i = m
  delocal int i = m

procedure pattern_hash(int P[][], int m, int d, int q, int ph)
  local int z1 = 0
    local int z2 = 0
      local int Vp = 0
        call compute_V(P, z1, z2, m, d, Vp)
        ph += Vp % q
        uncall compute_V(P, z1, z2, m, d, Vp)
      delocal int Vp = 0
    delocal int z2 = 0
  delocal int z1 = 0
