procedure rowsum(int A[][], int s1, int s2, int i, int m, int d, int rs)
  local int k = 0
  from k = 0 do
    rs *= d
    rs += A[s1+i][s2+k]
    k += 1
  loop skip
  until k = m
  delocal int k = m

procedure windowhash(int A[][], int s1, int s2, int m, int d, int q, int wh)
  local int ws = 0
    local int i = 0
    from i = 0 do
      ws *= d
      local int rs = 0
        call rowsum(A, s1, s2, i, m, d, rs)
        local int rh = 0
          rh += rs % q
          ws += rh
          rh -= rs % q
        delocal int rh = 0
        uncall rowsum(A, s1, s2, i, m, d, rs)
      delocal int rs = 0
      i += 1
    loop skip
    until i = m
    delocal int i = m
    wh += ws % q
    local int i2 = m
    from i2 = m do
      i2 -= 1
      local int rs = 0
        call rowsum(A, s1, s2, i2, m, d, rs)
        local int rh = 0
          rh += rs % q
          ws -= rh
          rh -= rs % q
        delocal int rh = 0
        uncall rowsum(A, s1, s2, i2, m, d, rs)
      delocal int rs = 0
      ws /= d
    loop skip
    until i2 = 0
    delocal int i2 = 0
  delocal int ws = 0
