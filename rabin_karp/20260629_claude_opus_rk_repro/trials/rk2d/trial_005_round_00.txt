procedure rowhash(int A[][], int row, int col, int len, int d, int h)
   local int k = 0
      from k = 0 do
         h *= d
         h += A[row][col + k]
      loop
         k += 1
      until k = len - 1
      delocal int k = len - 1

procedure windowhash(int A[][], int s1, int s2, int m, int d, int WH)
   local int i = 0
      from i = 0 do
         WH *= d
         local int WR = 0
            call rowhash(A, s1 + i, s2, m, d, WR)
            WH += WR
            uncall rowhash(A, s1 + i, s2, m, d, WR)
         delocal int WR = 0
      loop
         i += 1
      until i = m - 1
      delocal int i = m - 1

procedure countmism(int T[][], int s1, int s2, int P[][], int m, int mism)
   local int i = 0
      from i = 0 do
         local int j = 0
            from j = 0 do
               if T[s1 + i][s2 + j] != P[i][j] then
                  mism += 1
               else
                  skip
               fi T[s1 + i][s2 + j] != P[i][j]
            loop
               j += 1
            until j = m - 1
            delocal int j = m - 1
      loop
         i += 1
      until i = m - 1
      delocal int i = m - 1

procedure rabinkarp2(int T[][], int n, int P[][], int m, int d, int q, stack r)
   local int PH = 0
      call windowhash(P, 0, 0, m, d, PH)
      local int s2 = 0
         from s2 = 0 do
            local int s1 = 0
               from s1 = 0 do
                  local int WH = 0
                     call windowhash(T, s1, s2, m, d, WH)
                     if (WH % q) = (PH % q) then
                        local int mism = 0
                           call countmism(T, s1, s2, P, m, mism)
                           if mism = 0 then
                              local int a = s1
                                 push(a, r)
                              delocal int a = 0
                              local int b = s2
                                 push(b, r)
                              delocal int b = 0
                           else
                              skip
                           fi mism = 0
                           uncall countmism(T, s1, s2, P, m, mism)
                        delocal int mism = 0
                     else
                        skip
                     fi (WH % q) = (PH % q)
                     uncall windowhash(T, s1, s2, m, d, WH)
                  delocal int WH = 0
               loop
                  s1 += 1
               until s1 = n - m
               delocal int s1 = n - m
         loop
            s2 += 1
         until s2 = n - m
         delocal int s2 = n - m
      uncall windowhash(P, 0, 0, m, d, PH)
   delocal int PH = 0
