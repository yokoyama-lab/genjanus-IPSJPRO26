The error is simply that `PHmod` is never uncomputed before its `delocal` (it passed case 1 only because the pattern hash happened to be 0 there). I add a matching uncompute of `PHmod` at the end.

procedure rowfull(int A[][], int row, int col, int m, int d, int h)
   local int j = 0
      from j = 0 do
         h *= d
         h += A[row][col + j]
      loop
         j += 1
      until j = m - 1
      delocal int j = m - 1

procedure rhcombine(int RH[], int nm1, int s2, int s1, int m, int d, int g)
   local int i = 0
      from i = 0 do
         g *= d
         g += RH[(s1 + i) * nm1 + s2]
      loop
         i += 1
      until i = m - 1
      delocal int i = m - 1

procedure pcombine(int A[][], int m, int d, int q, int g)
   local int i = 0
      from i = 0 do
         g *= d
         local int h = 0
            call rowfull(A, i, 0, m, d, h)
            g += h % q
            uncall rowfull(A, i, 0, m, d, h)
         delocal int h = 0
      loop
         i += 1
      until i = m - 1
      delocal int i = m - 1

               c += 1
            until c = nm1 - 1
            delocal int c = nm1 - 1
      loop
         r += 1
      until r = n - 1
      delocal int r = n - 1
