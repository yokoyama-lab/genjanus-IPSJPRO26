Looking at the repeated rejection: my hash math was hidden in helpers and (in the recursive version) buried away from the visible window scan, and the exact-roll versions never reduced mod q each step. I'm switching to the explicitly-requested FORWARD-ONLY array design: a local array holds the per-window modular hashes along the s3 axis, each new cell filled with a single reversible `+=` of the modular rolling update (`prev*D + entering - leaving*DT`, reduced `% q`, no inverse of d), and cleaned up by re-subtracting the identical forward expression.

procedure power(int d, int e, int pw)
  local int c = 0
  from c = 0 do
    skip
  loop
    pw *= d
    c += 1
  until c = e
  delocal int c = e

procedure addslicehash(int T[][][], int s1, int s2, int p, int m, int d, int q, int acc)
  local int i = 0
  from i = 0 do
    local int j = 0
    from j = 0 do
      local int e = i * m + j
        local int pw = 1
          call power(d, e, pw)
          acc += ((T[s1 + i][s2 + j][p] % q) * (pw % q)) % q
          uncall power(d, e, pw)
        delocal int pw = 1
      delocal int e = i * m + j
      j += 1
    loop
      skip
    until j = m
    delocal int j = m
    i += 1
  loop
    skip
  until i = m
  delocal int i = m

procedure basehash(int T[][][], int s1, int s2, int m, int d, int q, int mm, int acc)
  local int c = 0
  from c = 0 do
    local int sl = 0
      call addslicehash(T, s1, s2, c, m, d, q, sl)
      local int e = mm * (m - 1 - c)
        local int pw = 1
          call power(d, e, pw)
          acc += ((sl % q) * (pw % q)) % q
          uncall power(d, e, pw)
        delocal int pw = 1
      delocal int e = mm * (m - 1 - c)
      uncall addslicehash(T, s1, s2, c, m, d, q, sl)
    delocal int sl = 0
    c += 1
  loop
    skip
  until c = m
  delocal int c = m

procedure pushpos(int s1, int s2, int s3, stack r)

  delocal int a = 0
