procedure cubehash(int A[][][], int bi, int bj, int bk, int m, int d, int h)
  local int i = 0
  from i = 0 do
    skip
  loop
    local int j = 0
    from j = 0 do
      skip
    loop
      local int k = 0
      from k = 0 do
        skip
      loop
        h *= d
        h += A[bi+i][bj+j][bk+k]
        k += 1
      until k = m
      delocal int k = m
      j += 1
    until j = m
    delocal int j = m
    i += 1
  until i = m
  delocal int i = m
