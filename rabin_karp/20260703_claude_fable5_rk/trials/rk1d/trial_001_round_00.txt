procedure pdm(int res, int d, int q, int k)
    if k = 0 then
        res += 1 % q
    else
        local int k1 = k - 1
        local int sub = 0
        call pdm(sub, d, q, k1)
        res += (sub * d) % q
        uncall pdm(sub, d, q, k1)
        delocal int sub = 0
        delocal int k1 = k - 1
    fi k = 0
