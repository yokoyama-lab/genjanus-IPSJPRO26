    pw[0] += 1
    local int k = 0
    from k = 0 do
        local int t = (pw[k] * b) % q
        pw[k+1] += t
        delocal int t = (pw[k] * b) % q
    loop
        k += 1
    until k = cnt - 1
    delocal int k = cnt - 1
