procedure pows(int W[], int b, int m, int q)
    W[0] += 1
    local int t = 0
    from t = 0 do
        local int u = 0
        u += ((W[t] * b) % q)
        W[t+1] += u
        u -= ((W[t] * b) % q)
        delocal int u = 0
        t += 1
    loop skip
    until t = m
    delocal int t = m
