procedure fillpow(int DP[], int m, int d, int q)
    DP[0] += 1
    local int k = 0
    from k = 0 do
        local int t = (DP[k] * d) % q
        DP[k + 1] += t
        delocal int t = DP[k + 1]
        k += 1
    loop skip until k = m
    delocal int k = m

procedure wseed3(int A[][][], int i, int j, int m, int d, int q, int W[])

    from k = 0 do
        local int t = (W[k] * d + A[i][j][k]) % q

        delocal int t = W[k + 1]
        k += 1
    loop skip until k = m
    delocal int k = m

procedure wseed1(int A[], int off, int step, int m, int d, int q, int W[])

procedure phash(int P[][][], int m, int d, int q, int W[], int PZ[], int PY[], int PH[])

    from i = 0 do
        local int j = 0
        from j = 0 do

            uncall wseed3(P, i, j, m, d, q, W)
            j += 1
        loop skip until j = m
        delocal int j = m
        i += 1
    loop skip until i = m
    delocal int i = m

procedure diff(int T[][][], int P[][][], int s1, int s2, int s3, int m, int DF[])

        local int j = 0
        from j = 0 do
            local int k = 0

                    DF[0] += 1
                else
                    skip
                fi T[s1 + i][s2 + j][s3 + k] != P[i][j][k]
