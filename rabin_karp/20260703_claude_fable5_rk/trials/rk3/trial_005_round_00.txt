procedure powfill(int PW[], int rad, int m, int q)
    PW[0] += 1
    local int t = 0
    from t = 0 do
        local int x = (PW[t] * rad) % q
        PW[t + 1] += x
        delocal int x = PW[t + 1]
        t += 1
    until t = m
    delocal int t = m
